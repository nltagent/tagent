# Единственная зависимость во всём проекте. Модуль stdlib zoneinfo
# (используется в modules/reminders/timeparse.py для часовых поясов
# и перевода летнего/зимнего времени) полагается на системную базу
# IANA tz — она не гарантированно есть в python:3.12-slim. tzdata —
# официальный pure-Python пакет от Python core team именно для этого
# случая: без скомпилированных зависимостей, не тянет ничего за собой.
#
# Больше внешних зависимостей нет: все HTTP-вызовы — через urllib,
# все данные — через sqlite3, всё остальное — стандартная библиотека.
#
# Если сюда всё же добавится ещё пакет — фиксируем точную версию (==),
# без диапазонов (>=, ~=), чтобы пересборка образа не подтягивала
# неожиданно новый мажорный/минорный релиз.
tzdata==2026.2
